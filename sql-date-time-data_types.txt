
create table emp(
	name varchar(20),
    age int,
    joining_date date default (current_date()),
    work_hours time
);

select * from emp;

insert into emp (name, age, work_hours) values("helo", 26, '4:5:8');

drop table emp;



create table emp_login(
	name varchar(20),
    login_time datetime default current_timestamp,
    login_new_time timestamp default current_timestamp,
	joining_year year
);

select * from emp_login;

insert into emp_login (name, joining_year) values("kris", 2025);

drop table emp_login;
