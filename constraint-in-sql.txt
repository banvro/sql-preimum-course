	name varchar(30) not null,
    email varchar(150) not null,
    phone_number varchar(12)
);

create table insta(
	id int auto_increment unique,
    username varchar(30),
    email varchar(150),
    full_name varchar(100),
    
    constraint u_email_username unique (username, email)
);

describe insta;

insert into insta (username, email, full_name) values
("x", "x@gmail.com", "z");

select * from insta;

alter table insta drop constraint unq_email;

drop table insta;

create table voat(
	id int auto_increment unique,
    name varchar(30) not null,
    age int,
    username varchar(30),
    
    constraint eligible check (age > 18),
    constraint check_usename check (length(username) < 10 
    and length(username) > 3)
);

insert into voat (name, age, username) values ("x", 20, "hloe");



drop table voat;

create table emp(
	id int auto_increment unique,
    emp_name varchar(30) not null,
    emp_email varchar(50) unique not null,
    age int check (age > 20),
    attendence varchar(10) default "absent"
);
